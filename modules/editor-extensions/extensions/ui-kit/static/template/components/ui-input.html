<ui-input @change="onChange($event.target.value)" @confirm="onConfirm($event.target.value)" @cancel="onCancel($event.target.value)" ></ui-input> // vue object methods = { onConfirm(value: string) { console.log('onConfirm', value); }, onChange(value: string) { console.log('onChange', value); }, onCancel(value: string) { console.log('onCancel', value); } };
