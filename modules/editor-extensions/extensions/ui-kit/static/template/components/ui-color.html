Events

change confirm cancel // ESC cancel, restore the last value

VUE JS

<ui-color ref="color" value="{{value}}" @confirm="onConfirm($event.target.value)" @change="onChange($event.target.value)" @cancel="onCancel($event.target.value)" ></ui-color>

Vanilla JS

<ui-color id="color" value="{{value}}"></ui-color> const element = document.getElementById('#color'); element.addEventListener('confirm', (event) => { const colorValue = event.target.value; }); element.addEventListener('change', (event) => { console.log('onChange', event.target.value); }); element.addEventListener('cancel', (event) => { console.log('onCancel', event.target.value); });

const element = document.getElementById('#color'); // set value element.value = '{{value}}'; // get value const colorValue = element.value;
